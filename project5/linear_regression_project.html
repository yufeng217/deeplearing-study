def is_near_0(n, epsilon=1.0e-16):
    return abs(n - 0.0) <= epsilon

def is_near_1(n, epsilon=1.0e-16):
    return abs(n - 1.0) <= epsilon

def scaleRow(M, r, scale, epsilon=1.0e-16):
    #print("row : {}, scale : {}".format(r, scale))
    if scale == 0:
        raise ValueError
    else:
        M[r] = [1.0 if is_near_1(x * scale, epsilon) else x * scale for x in M[r]]

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    # 1 Ab
    if shape(A)[0] != shape(b)[0]:
        return None
    # 2 Ab
    Ab = augmentMatrix(A, b)
    # 3 Ab
    for i in range(len(Ab)):
        col = transpose(Ab)[i][i:]
        #print(col)
        col_max = max(map(abs, col))
        if col_max in col:
            ri_max = col.index(col_max) + i
        else:
            ri_max = col.index(-col_max) + i
        if is_near_0(col_max, epsilon):
            col_max = 0.0
        if col_max == 0.0:
            # A
            return None
        else:
            # c
            swapRows(Ab, i, ri_max)
            # c1
            scaleRow(Ab, i, 1./Ab[i][i], epsilon)
            # c0
            # 
            current_col = transpose(Ab)[i]
            for c_i, c_x in enumerate(current_col):
                if c_i != i:
                    addScaledRow(Ab, c_i, i, -c_x, epsilon)
    result = [[x[-1]] for x in Ab]
    return result
